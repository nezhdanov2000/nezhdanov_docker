======================================== ЗАДАНИЕ 1 =================================================
#СОЗДАДИМ НОВЫЙ ОБРАЗ
docker build -t nezhdanov_image:latest .
---------------------------------------------------------------------------------------------------
#ЗАПУСТИМ КОНТЕЙНЕР
docker run -d -p 5432:5432 --name nezhdanov_container nezhdanov_image:latest
